15  package org.basepom.mojo.propertyhelper.definitions;
16  
17  import static com.google.common.base.Preconditions.checkState;
18  
19  import org.basepom.mojo.propertyhelper.Field;
20  import org.basepom.mojo.propertyhelper.FieldContext;
21  import org.basepom.mojo.propertyhelper.IgnoreWarnFailCreate;
22  import org.basepom.mojo.propertyhelper.ValueCache;
23  
24  import java.io.File;
25  import java.io.IOException;
26  import java.util.List;
27  import java.util.Objects;
28  import java.util.Optional;
29  import java.util.StringJoiner;
30  import java.util.function.Function;
31  
32  import com.google.common.annotations.VisibleForTesting;
33  import com.google.common.base.Splitter;
34  
35  /**
36   * Common properties for a field.
37   */
38  public abstract class FieldDefinition<T> {
39  
40      protected FieldDefinition() {
41      }
42  
43      @VisibleForTesting
44      protected FieldDefinition(String id) {
45          this.id = id;
46      }
47  
48      /**
49       * Name of the build property to define. Field injected by Maven.
50       */
51      String id = null;
52  
53      /**
54       * True skips the parsing of this definition. Field injected by Maven.
55       */
56      boolean skip = false;
57  
58      /**
59       * Whether to export this number directly. Field injected by Maven.
60       */
61      boolean export = false;
62  
63      /**
64       * Name of the property from the properties file. Field injected by Maven.
65       */
66      String propertyNameInFile = null;
67  
68      /**
69       * Name of the properties file to persist the count. Field injected by Maven.
70       */
71      File propertyFile = null;
72  
73      /**
74       * What to do when the property is missing from the file.
75       */
76      private IgnoreWarnFailCreate onMissingFile = IgnoreWarnFailCreate.FAIL;
77  
78      public void setOnMissingFile(String onMissingFile) {
79          this.onMissingFile = IgnoreWarnFailCreate.forString(onMissingFile);
80      }
81  
82      /**
83       * What to do when the property is missing from the file.
84       */
85      private IgnoreWarnFailCreate onMissingProperty = IgnoreWarnFailCreate.FAIL;
86  
87      public void setOnMissingProperty(String onMissingProperty) {
88          this.onMissingProperty = IgnoreWarnFailCreate.forString(onMissingProperty);
89      }
90  
91      /**
92       * The initial value for this field. Field injected by Maven.
93       */
94      String initialValue = null;
95  
96      /**
97       * Format for this element. Field injected by Maven.
98       */
99      String format = null;
100 
101     /**
102      * Comma separated list of String transformers.
103      */
104     private List<String> transformers = List.of();
105 
106     void setTransformers(String transformers) {
107         this.transformers = Splitter.on(",")
108             .omitEmptyStrings()
109             .trimResults()
110             .splitToList(transformers);
111     }
112 
113     public abstract <U extends Field<?, ?>> U createField(FieldContext context, ValueCache valueCache) throws IOException;
114 
115     public String getId() {
116         return id;
117     }
118 
119     public boolean isSkip() {
120         return skip;
121     }
122 
123     public List<String> getTransformers() {
124         return transformers;
125     }
126 
127     public Optional<String> getInitialValue() {
128         return Optional.ofNullable(initialValue);
129     }
130 
131     public boolean isExport() {
132         return export;
133     }
134 
135     public String getPropertyNameInFile() {
136         return Objects.requireNonNullElse(propertyNameInFile, id);
137     }
138 
139     public Optional<File> getPropertyFile() {
140         return Optional.ofNullable(propertyFile);
141     }
142 
143     public IgnoreWarnFailCreate getOnMissingFile() {
144         return onMissingFile;
145     }
146 
147     public IgnoreWarnFailCreate getOnMissingProperty() {
148         return onMissingProperty;
149     }
150 
151     public Function<T, String> formatResult() {
152         return value -> Optional.ofNullable(value).map(getFormat()).orElse("");
153     }
154 
155     public Function<T, String> getFormat() {
156         return format == null ? String::valueOf : v -> String.format(format, v);
157     }
158 
159     public void check() {
160         checkState(id != null, "the id element must not be empty!");
161 
162         if (propertyNameInFile != null) {
163             checkState(propertyFile != null, "can not define <propertyNameInFile> without defining <propertyFile>!");
164         }
165     }
166 
167     @Override
168     public String toString() {
169         return new StringJoiner(", ", FieldDefinition.class.getSimpleName() + "[", "]")
170             .add("id='" + id + "'")
171             .add("skip=" + skip)
172             .add("export=" + export)
173             .add("propertyNameInFile='" + propertyNameInFile + "'")
174             .add("propertyFile=" + propertyFile)
175             .add("onMissingFile='" + onMissingFile + "'")
176             .add("onMissingProperty='" + onMissingProperty + "'")
177             .add("initialValue='" + initialValue + "'")
178             .add("format='" + format + "'")
179             .add("transformers='" + transformers + "'")
180             .toString();
181     }
182 
183     @Override
184     public boolean equals(Object o) {
185         if (this == o) {
186             return true;
187         }
188         if (o == null || getClass() != o.getClass()) {
189             return false;
190         }
191         FieldDefinition<?> that = (FieldDefinition<?>) o;
192         return skip == that.skip && export == that.export && Objects.equals(id, that.id) && Objects.equals(propertyNameInFile, that.propertyNameInFile)
193             && Objects.equals(propertyFile, that.propertyFile) && Objects.equals(onMissingFile, that.onMissingFile)
194             && Objects.equals(onMissingProperty, that.onMissingProperty) && Objects.equals(initialValue, that.initialValue)
195             && Objects.equals(format, that.format) && Objects.equals(transformers, that.transformers);
196     }
197 
198     @Override
199     public int hashCode() {
200         return Objects.hash(id, skip, export, propertyNameInFile, propertyFile, onMissingFile, onMissingProperty, initialValue, format, transformers);
201     }
202 }
