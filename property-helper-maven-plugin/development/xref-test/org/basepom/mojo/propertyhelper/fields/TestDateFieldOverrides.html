14  package org.basepom.mojo.propertyhelper.fields;
15  
16  import static org.assertj.core.api.Assertions.assertThat;
17  import static org.basepom.mojo.propertyhelper.definitions.DefinitionHelper.dateDefinition;
18  import static org.basepom.mojo.propertyhelper.definitions.DefinitionHelper.setInitialValue;
19  import static org.basepom.mojo.propertyhelper.definitions.DefinitionHelper.setValue;
20  
21  import org.basepom.mojo.propertyhelper.FieldContext;
22  import org.basepom.mojo.propertyhelper.ValueCache;
23  import org.basepom.mojo.propertyhelper.ValueProvider.SingleValueProvider;
24  import org.basepom.mojo.propertyhelper.definitions.DateDefinition;
25  
26  import java.time.Instant;
27  import java.time.ZoneId;
28  import java.time.ZonedDateTime;
29  import java.time.format.DateTimeFormatter;
30  
31  import org.junit.jupiter.api.Test;
32  
33  class TestDateFieldOverrides {
34  
35      @Test
36      void testZeroValue() throws Exception {
37          DateDefinition dateDefinition = dateDefinition("hello");
38          setInitialValue(dateDefinition, "0");
39  
40          ZonedDateTime localEpochTime = ZonedDateTime.ofInstant(Instant.ofEpochMilli(0), ZoneId.systemDefault());
41  
42          var context = FieldContext.EMPTY_CONTEXT;
43          ValueCache valueCache = new ValueCache();
44          var dateField = dateDefinition.createField(context, valueCache);
45          var value = dateField.getValue();
46  
47          assertThat(value).isEqualTo(DateTimeFormatter.ISO_DATE_TIME.format(localEpochTime));
48      }
49  
50      @Test
51      void testLoadedValueOverridesInitialValue() throws Exception {
52          DateDefinition dateDefinition = dateDefinition("hello");
53          setInitialValue(dateDefinition, "0");
54  
55          ZonedDateTime localEpochPlusOneTime = ZonedDateTime.ofInstant(Instant.ofEpochMilli(1), ZoneId.systemDefault());
56  
57          var valueProvider = new SingleValueProvider();
58          valueProvider.setValue("1");
59          var dateField = DateField.forTesting(dateDefinition, valueProvider);
60  
61          var value = dateField.getValue();
62  
63          assertThat(value).isEqualTo(DateTimeFormatter.ISO_DATE_TIME.format(localEpochPlusOneTime));
64      }
65  }
